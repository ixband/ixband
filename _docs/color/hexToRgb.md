### Example
```js
//Return Type : String
var strColor = $B.color.hexToRgb( '#FFFFFF' ); //rgb(255, 255, 255)

//Return Type : Object
var objColor = $B.color.hexToRgb( '#FFFFFF', true ); //{ r: 255, g: 255, b: 255 }
```
